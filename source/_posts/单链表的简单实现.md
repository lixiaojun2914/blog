# 单链表的简单实现
1. 结点的定义
```c++
typedef struct LNode{
    int data;
    LNode *next;
}LNode, *LinkList;
```
2. 初始化头结点
```c++
void InitList(LinkList &L){
    L = new LNode();
    L->next = NULL;
}
```
3. 链表的插入
```c++
void InsertList(LinkList &L, int e){
    LinkList p = L;
    while(p->next!=NULL) p=p->next;
    LNode *q = new LNode();
    q->data = e;
    q->next = p->next;
    p->next = q;
}
```
4. 链表的打印
```c++
void PrintList(LinkList L){
    if(L->next == NULL) return;
    LNode *p = L->next;
    while(p){
        printf("%d ", p->data);
        p=p->next;
    }
    printf("\n");
}
```
5. 链表的翻转
```c++
void ReverseList(LinkList &L){
    LinkList p = L->next;
    if(p->next==NULL) return;
    L->next=NULL;
    LinkList q = p->next;
    p->next=NULL;
    while(q){
        LinkList r = q->next;
        q->next=p;
        p = q;
        q = r;
    }
    L->next = p;
}
```
6. 测试用例
```c++
int main(){
    LinkList L;
    InitList(L);
    InsertList(L, 1);
    InsertList(L, 2);
    InsertList(L, 3);
    InsertList(L, 4);
    InsertList(L, 5);
    PrintList(L);
    ReverseList(L);
    PrintList(L);
    return 0;
}
```
